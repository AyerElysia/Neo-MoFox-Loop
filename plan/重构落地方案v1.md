```python
    STABLE = "stable"       # 慢变量层


@dataclass(frozen=True)
class StateUpdate:
    """状态更新记录"""
    key: str
    old_value: Any
    new_value: Any
    source: str           # 更新来源 (plugin_name:component:action)
    reason: str           # 更新原因
    timestamp: datetime


class StateVariable(ABC, Generic[T]):
    """状态变量基类"""
    
    @property
    @abstractmethod
    def key(self) -> str:
        """状态键名"""
        ...
    
    @property
    @abstractmethod
    def layer(self) -> StateLayer:
        """所属层级"""
        ...
    
    @property
    @abstractmethod
    def value(self) -> T:
        """当前值"""
        ...
    
    @abstractmethod
    def update(self, new_value: T, source: str, reason: str) -> StateUpdate:
        """更新值"""
        ...
    
    @abstractmethod
    def decay(self, factor: float) -> None:
        """衰减（仅快变量层）"""
        ...
    
    @abstractmethod
    def to_dict(self) -> dict[str, Any]:
        """序列化"""
        ...
    
    @classmethod
    @abstractmethod
    def from_dict(cls, data: dict[str, Any]) -> "StateVariable[T]":
        """反序列化"""
        ...
```

```python
@dataclass
class EmotionVector:

    arousal: float = 0.0      # 唤醒度 (0 ~ 1): 平静 ~ 激动

        self.dominance = max(0.0, min(1.0, self.dominance))

    """情绪状态变量"""

        self._value = initial or EmotionVector()
        self._history: list[StateUpdate] = []

        self._history.append(update)

            },
```

#### 3.1.3 桥接层实现

```python
# src/core/state/layers/bridge.py

from dataclasses import dataclass
from typing import Callable, Any

from ..protocols import StateVariable, StateLayer


@dataclass
class BridgeRule:
    """桥接规则：快变量 → 慢变量的沉淀条件"""
    source_key: str               # 快变量键
    target_key: str               # 慢变量键
    threshold: float              # 触发阈值
    accumulator: str              # 累积方式 ("sum", "avg", "max")
    decay_factor: float = 0.1    # 触发后衰减因子
    min_occurrences: int = 3     # 最小触发次数
    
    condition: Callable[[Any], bool] | None = None  # 自定义触发条件


class BridgeLayer:
    """桥接层：管理快变量到慢变量的沉淀"""
    
    def __init__(self) -> None:
        self._rules: dict[str, list[BridgeRule]] = {}
        self._accumulators: dict[str, list[Any]] = {}
    
    def register_rule(self, rule: BridgeRule) -> None:
        """注册桥接规则"""
        if rule.source_key not in self._rules:
            self._rules[rule.source_key] = []
        self._rules[rule.source_key].append(rule)
        self._accumulators[f"{rule.source_key}:{rule.target_key}"] = []
    
    def on_volatile_update(
        self, 
        key: str, 
        value: Any,
        state_system: "UnifiedStateSystem"
    ) -> list[str]:
        """处理快变量更新，检查是否触发沉淀"""
        triggered_targets: list[str] = []
        
        if key not in self._rules:
            return triggered_targets
        
        for rule in self._rules[key]:
            acc_key = f"{rule.source_key}:{rule.target_key}"
            self._accumulators[acc_key].append(value)
            
            # 检查触发条件
            if len(self._accumulators[acc_key]) >= rule.min_occurrences:
                should_trigger = self._check_threshold(
                    self._accumulators[acc_key],
                    rule.threshold,
                    rule.accumulator,
                    rule.condition,
                )
                
                if should_trigger:
                    # 触发沉淀
                    self._settle_to_stable(
                        rule.target_key,
                        self._accumulators[acc_key],
                        state_system,
                    )
                    # 衰减快变量
                    state_system.decay_variable(key, rule.decay_factor)
                    # 清空累积器
                    self._accumulators[acc_key] = []
                    triggered_targets.append(rule.target_key)
        
        return triggered_targets
    
    def _check_threshold(
        self, 
        values: list[Any], 
        threshold: float,
        accumulator: str,
        condition: Callable[[Any], bool] | None,
    ) -> bool:
        """检查是否达到阈值"""
        if condition:
            return condition(values)
        
        # 数值型累积
        numeric_values = [v for v in values if isinstance(v, (int, float))]
        if not numeric_values:
            return False
        
        if accumulator == "sum":
            return sum(numeric_values) >= threshold
        elif accumulator == "avg":
            return sum(numeric_values) / len(numeric_values) >= threshold
        elif accumulator == "max":
            return max(numeric_values) >= threshold
        return False
    
    def _settle_to_stable(
        self,
        target_key: str,
        values: list[Any],
        state_system: "UnifiedStateSystem",
    ) -> None:
        """沉淀到慢变量"""
        # 计算沉淀值（简化示例：取平均）
        numeric_values = [v for v in values if isinstance(v, (int, float))]
        if numeric_values:
            settled_value = sum(numeric_values) / len(numeric_values)
            state_system.update_state(
                target_key,
                settled_value,
                source="bridge_layer",
                reason=f"Settled from accumulator (n={len(values)})",
            )
```

```python
# src/core/scheduler/central.py

from dataclasses import dataclass
from typing import Any, Callable
from enum import Enum

from .task_mode.base import TaskMode, TaskModeType, ModeTransition, ModeResult
from .registry.task_registry import TaskRegistry
from .registry.agent_registry import AgentRegistry
from .trigger import SignalTrigger, Signal
from ..state import UnifiedStateSystem


class ScheduleDecision(str, Enum):
    """调度决策"""
    STAY = "stay"           # 保持当前态
    SWITCH = "switch"       # 切换态
    DEFER = "defer"         # 延迟处理
    DELEGATE = "delegate"   # 委托给 Agent


@dataclass
class AgentTask:
    """Agent 任务"""
    task_id: str
    agent_id: str
    request: Any
    callback: Callable[[Any], None] | None = None


class UnifiedScheduler:
    """统一调度中枢"""
    
    def __init__(
        self,
        state_system: UnifiedStateSystem,
        task_registry: TaskRegistry | None = None,
        agent_registry: AgentRegistry | None = None,
    ) -> None:
        self._state_system = state_system
        self._task_registry = task_registry or TaskRegistry()
        self._agent_registry = agent_registry or AgentRegistry()
        self._trigger = SignalTrigger(state_system)
        
        self._current_mode: TaskMode | None = None
        self._pending_tasks: dict[str, AgentTask] = {}
        self._mode_history: list[tuple[TaskModeType, str]] = []
        
        # 注册默认任务态
        self._register_default_modes()
    
    def _register_default_modes(self) -> None:
        """注册默认任务态"""
        from .task_mode.chat import ChatMode
        from .task_mode.think import ThinkMode
        
        self._task_registry.register(ChatMode())
        self._task_registry.register(ThinkMode())
    
    @property
    def current_mode(self) -> TaskMode | None:
        return self._current_mode
    
    @property
    def current_mode_type(self) -> TaskModeType | None:
        return self._current_mode.mode_type if self._current_mode else None
    
    async def initialize(self, initial_mode: TaskModeType = TaskModeType.CHAT) -> None:
        """初始化调度器"""
        mode = self._task_registry.get(initial_mode)
        if mode:
            self._current_mode = mode
            await mode.enter(ModeTransition(
                from_mode=initial_mode,
                to_mode=initial_mode,
                reason="Initial startup",
                context_slice=self._state_system.get_slice_for_mode(initial_mode),
            ))
    
    def evaluate_signals(self, signals: list[Signal]) -> tuple[ScheduleDecision, Any]:
        """评估信号，决定调度策略"""
        if not signals:
            return ScheduleDecision.STAY, None
        
        # 优先级：高不确定性 > 上下文冲突 > 复杂问题 > 常规
        for signal in sorted(signals, key=lambda s: s.priority, reverse=True):
            # 检查是否需要切换态
            if signal.type == "high_uncertainty" and signal.value > 0.8:
                return ScheduleDecision.SWITCH, TaskModeType.THINK
            
            if signal.type == "context_conflict":
                return ScheduleDecision.SWITCH, TaskModeType.THINK
            
            if signal.type == "memory_gap":
                return ScheduleDecision.DELEGATE, "memory_agent"
            
            if signal.type == "resource_need":
                return ScheduleDecision.DELEGATE, "resource_agent"
        
        return ScheduleDecision.STAY, None
    
    async def switch_mode(
        self, 
        target: TaskModeType, 
        reason: str,
    ) -> bool:
        """切换任务态"""
        target_mode = self._task_registry.get(target)
        if not target_mode:
            return False
        
        # 记录历史
        if self._current_mode:
            self._mode_history.append((self._current_mode.mode_type, reason))
            await self._current_mode.exit()
        
        # 准备上下文转移
        context_slice = self._state_system.get_slice(target_mode.required_state_keys)
        
        # 进入新态
        transition = ModeTransition(
            from_mode=self._current_mode.mode_type if self._current_mode else target,
            to_mode=target,
            reason=reason,
            context_slice=context_slice,
        )
        
        await target_mode.enter(transition)
        self._current_mode = target_mode
        return True
    
    async def dispatch_to_agent(
        self, 
        agent_id: str, 
        request: Any,
        callback: Callable[[Any], None] | None = None,
    ) -> AgentTask | None:
        """分发任务给 Agent"""
        agent = self._agent_registry.get(agent_id)
        if not agent:
            return None
        
        import uuid
        task_id = str(uuid.uuid4())
        task = AgentTask(
            task_id=task_id,
            agent_id=agent_id,
            request=request,
            callback=callback,
        )
        self._pending_tasks[task_id] = task
        
        # 异步执行
        from src.kernel.concurrency import get_task_manager
        get_task_manager().create_task(
            self._execute_agent_task(task, agent),
            name=f"agent_task_{task_id}",
        )
        
        return task
    
    async def _execute_agent_task(self, task: AgentTask, agent: Any) -> None:
        """执行 Agent 任务"""
        try:
            result = await agent.execute(task.request)
            if task.callback:
                task.callback(result)
            # 通过门控决定如何注入主意识
            await self._gate_result_to_conscious(task, result)
        finally:
            self._pending_tasks.pop(task.task_id, None)
    
    async def _gate_result_to_conscious(self, task: AgentTask, result: Any) -> None:
        """门控：决定 Agent 结果如何注入主意识"""
        # 判断结果重要性
        importance = self._evaluate_result_importance(result)
        
        if importance > 0.8:
            # 高重要性：立即注入
            self._state_system.update_state(
                f"agent_result_{task.agent_id}",
                result,
                source=f"agent:{task.agent_id}",
                reason="High importance result",
            )
        elif importance > 0.5:
            # 中等重要性：等待主意识空闲时注入
            self._state_system.queue_injection(
                f"agent_result_{task.agent_id}",
                result,
            )
        else:
            # 低重要性：仅记录日志
            pass
    
    def _evaluate_result_importance(self, result: Any) -> float:
        """评估结果重要性"""
        # 简化实现
        return 0.5
    
    def get_agent(self, agent_id: str) -> Any | None:
        """获取 Agent 实例"""
        return self._agent_registry.get(agent_id)
    
    def register_mode(self, mode: TaskMode) -> None:
        """注册新任务态"""
        self._task_registry.register(mode)
    
    def register_agent(self, agent_id: str, agent: Any) -> None:
        """注册新 Agent"""
        self._agent_registry.register(agent_id, agent)
```

```python
        self, 
        stream_context: Any,
    ) -> AsyncGenerator["ChatterResult", None]:
        """处理对话，协调主意识与 Chatter"""
        # 1. 从主意识获取处理意图
        intent = await self._conscious.form_intent(stream_context)
        
        # 2. 检查是否需要深思
        if intent.needs_deep_thinking:
            # 通知调度器切换到深思态
            yield from self._delegate_to_scheduler(intent)
            return
        
        # 3. 常规对话处理
        async for result in self._chatter.execute(stream_context.unread_messages):
            # 4. 通过主意识整合结果
            integrated = await self._conscious.integrate_output(result)
            yield integrated
    
    async def _delegate_to_scheduler(self, intent: Any) -> AsyncGenerator[Any, None]:
        """委托给调度器处理"""
        from src.core.scheduler import get_scheduler
        scheduler = get_scheduler()
        await scheduler.switch_mode("think", reason=intent.reason)
        # 返回等待结果
        from src.core.components.base.chatter import Wait
        yield Wait(time=None)  # 无限等待直到深思完成
```
