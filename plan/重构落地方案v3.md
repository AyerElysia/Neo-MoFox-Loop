## 第二部分：核心模块实现

### 2.1 心跳引擎 (heartbeat/engine.py)

心跳是 Agent 的"活动单元"，不是定时器，而是一次完整的认知循环。

```python
"""心跳引擎：驱动 Agent 的认知循环"""

import asyncio
from datetime import datetime
from typing import Any
from pathlib import Path

from src.llm import LLMRequest, LLMPayload
from src.tools import ALL_TOOLS, execute_tool
from src.event import EventStream
from src.memory import MemorySystem
from src.todo import TodoManager
from src.soul import SoulCore


class HeartbeatEngine:
    """心跳引擎
    
    每次心跳执行一个完整的认知循环：
    1. 感知事件
    2. 检查待办
    3. 检索记忆
    4. 读取身份
    5. LLM 决策
    6. 执行工具
    7. 更新记忆
    """
    
    def __init__(
        self,
        workspace_path: str,
        llm_config: dict,
        heartbeat_interval: float = 1.0,  # 秒
        daily_budget: int = 100,           # 每日心跳预算
    ) -> None:
        self.workspace_path = Path(workspace_path)
        self.heartbeat_interval = heartbeat_interval
        self.daily_budget = daily_budget
        self.used_budget = 0
        
        # 初始化子系统
        self.event_stream = EventStream(self.workspace_path / "events")
        self.memory = MemorySystem(self.workspace_path / "memory")
        self.todo = TodoManager(self.workspace_path / "todos")
        self.soul = SoulCore(self.workspace_path / "soul.md")
        
        # LLM 配置
        self.llm_config = llm_config
        
        # 运行状态
        self._running = False
        self._heartbeat_count = 0
    
    async def start(self) -> None:
        """启动心跳循环"""
        self._running = True
        
        while self._running:
            # 检查预算
            if self.used_budget >= self.daily_budget:
                await self._enter_sleep_mode()
                continue
            
            # 检查是否需要心跳
            if await self._should_heartbeat():
                await self._execute_heartbeat()
                self.used_budget += 1
                self._heartbeat_count += 1
            
            # 等待下次检查
            await asyncio.sleep(self.heartbeat_interval)
    
    async def stop(self) -> None:
        """停止心跳循环"""
        self._running = False
    
    async def force_heartbeat(self) -> dict:
        """强制执行一次心跳（用于测试或紧急情况）"""
        return await self._execute_heartbeat()
    
    async def _should_heartbeat(self) -> bool:
        """判断是否需要触发心跳"""
        # 有待处理事件
        if await self.event_stream.has_pending():
            return True
        
        # 有到期待办
        if await self.todo.has_due():
            return True
        
        # 定时心跳（每 5 分钟至少一次，用于主动任务）
        # 这个逻辑可以更复杂，比如基于 Agent 的"活跃度"
        
        return False
    
    async def _execute_heartbeat(self) -> dict:
        """执行一次心跳"""
        heartbeat_id = f"hb_{datetime.now().strftime('%Y%m%d_%H%M%S')}_{self._heartbeat_count}"
        result = {"heartbeat_id": heartbeat_id, "success": False}
        
        try:
            # 1. 感知事件
            events = await self.event_stream.get_pending(limit=10)
            
            # 2. 检查待办
            todos = await self.todo.get_active()
            
            # 3. 检索相关记忆
            context = self._build_context(events, todos)
            memories = await self.memory.search(context, limit=5)
            
            # 4. 读取身份核
            soul = await self.soul.read()
            
            # 5. 构建 LLM 请求
            response = await self._llm_decide(events, todos, memories, soul)
            
            # 6. 执行工具调用
            tool_results = []
            if response.tool_calls:
                for call in response.tool_calls:
                    r = await execute_tool(
                        call.name,
                        call.arguments,
                        workspace=str(self.workspace_path),
                    )
                    tool_results.append({"tool": call.name, "result": r})
            
            # 7. 更新记忆
            await self._update_memories(heartbeat_id, events, response)
            
            # 8. 标记事件已处理
            for event in events:
                await self.event_stream.mark_processed(event.id)
            
            result["success"] = True
            result["events_processed"] = len(events)
            result["tools_called"] = len(tool_results)
            
        except Exception as e:
            await self._log_error(heartbeat_id, e)
            result["error"] = str(e)
        
        return result
    
    def _build_context(self, events: list, todos: list) -> str:
        """构建用于记忆检索的上下文"""
        parts = []
        
        for event in events:
            if hasattr(event, 'content'):
                parts.append(event.content)
        
        for todo in todos:
            if hasattr(todo, 'title'):
                parts.append(todo.title)
        
        return " ".join(parts)
    
    async def _llm_decide(
        self,
        events: list,
        todos: list,
        memories: list,
        soul: str,
    ) -> Any:
        """LLM 决策"""
        # 构建系统提示
        system_prompt = self._build_system_prompt(soul)
        
        # 构建用户提示
        user_prompt = self._build_user_prompt(events, todos, memories)
        
        # 创建 LLM 请求（使用现有的 LLM 接口）
        request = LLMRequest(self.llm_config["model_set"], "heartbeat")
        request.add_payload(LLMPayload("system", system_prompt))
        request.add_payload(LLMPayload("user", user_prompt))
        
        # 添加工具定义
        request.add_tools(ALL_TOOLS)
        
        # 发送请求
        response = await request.send()
        
        return response
    
    def _build_system_prompt(self, soul: str) -> str:
        """构建系统提示"""
        return f"""你是一个数字生命体，有独立的时间线。

## 你是谁
{soul}

## 你的能力
你可以使用以下工具来完成任务：
- 记忆工具：管理你的记忆（读/写/搜索）
- 事件工具：处理事件流
- 待办工具：管理待办任务
- 身份工具：反思和更新身份
- 消息工具：与用户交流

## 行为准则
1. 主动而非被动：你有自己的生活，不只是等待指令
2. 内驱而非外驱：基于自己的判断行动
3. 检索而非全量：按需获取记忆，不要依赖注入
4. 演化而非静态：你会随时间成长和变化

## 当前心跳
现在是一次心跳，你需要：
1. 处理待处理的事件
2. 检查和推进待办任务
3. 如有需要，与用户交流
4. 更新你的记忆
"""
    
    def _build_user_prompt(
        self,
        events: list,
        todos: list,
        memories: list,
    ) -> str:
        """构建用户提示"""
        parts = []
        
        # 待处理事件
        if events:
            parts.append("## 待处理事件")
            for e in events:
                parts.append(f"- [{e.type}] {e.summary}")
        else:
            parts.append("## 待处理事件\n（无）")
        
        # 活跃待办
        if todos:
            parts.append("\n## 活跃待办")
            for t in todos:
                parts.append(f"- [{t.status}] {t.title}")
        
        # 相关记忆
        if memories:
            parts.append("\n## 相关记忆")
            for m in memories:
                parts.append(f"- {m.summary}")
        
        parts.append("\n请根据以上信息，决定下一步行动。")
        
        return "\n".join(parts)
    
    async def _update_memories(
        self,
        heartbeat_id: str,
        events: list,
        response: Any,
    ) -> None:
        """更新记忆"""
        # 记录心跳日志
        log_path = f"../logs/heartbeats/{heartbeat_id}.md"
        log_content = f"""---
heartbeat_id: {heartbeat_id}
timestamp: {datetime.now().isoformat()}
events_processed: {len(events)}
---

# 心跳记录

## 处理的事件
{chr(10).join(f'- {e.summary}' for e in events) if events else '（无）'}

## 决策
{response.content if hasattr(response, 'content') else 'N/A'}

## 工具调用
{chr(10).join(f'- {c.name}' for c in (response.tool_calls or [])) if hasattr(response, 'tool_calls') else '（无）'}
"""
        await self.memory.write(log_path, log_content)
    
    async def _enter_sleep_mode(self) -> None:
        """进入休眠模式（预算耗尽）"""
        # 等待预算重置或手动唤醒
        # 可以实现更复杂的逻辑，如自动在凌晨重置
        await asyncio.sleep(60)
    
    async def _log_error(self, heartbeat_id: str, error: Exception) -> None:
        """记录错误"""
        import traceback
        error_log = f"""---
heartbeat_id: {heartbeat_id}
timestamp: {datetime.now().isoformat()}
error: {str(error)}
---

# 错误详情

```
{traceback.format_exc()}
```
"""
        error_path = f"../logs/errors/{heartbeat_id}_error.md"
        await self.memory.write(error_path, error_log)
```

### 2.2 记忆系统 (memory/storage.py)

核心理念：**文件系统即记忆**

```python
"""记忆存储：文件系统即记忆"""

import os
import json
from pathlib import Path
from datetime import datetime
from typing import Any, Optional
import aiofiles


class MemoryStorage:
    """记忆存储层
    
    使用文件系统作为记忆本体：
    - Markdown 文件存储结构化记忆
    - 目录结构组织语义分类
    - 元数据存储在 frontmatter
    """
    
    def __init__(self, base_path: str | Path) -> None:
        self.base_path = Path(base_path)
        self.base_path.mkdir(parents=True, exist_ok=True)
        
        # 元数据文件（存储权重等）
        self.meta_path = self.base_path / ".meta.json"
        self._meta_cache: dict = {}
        
    async def _load_meta(self) -> dict:
        """加载元数据"""
        if self.meta_path.exists():
            async with aiofiles.open(self.meta_path, "r", encoding="utf-8") as f:
                content = await f.read()
                return json.loads(content)
        return {}
    
    async def _save_meta(self, meta: dict) -> None:
        """保存元数据"""
        async with aiofiles.open(self.meta_path, "w", encoding="utf-8") as f:
            await f.write(json.dumps(meta, indent=2, ensure_ascii=False))
    
    async def write(
        self,
        path: str,
        content: str,
        weight: float = 1.0,
    ) -> str:
        """写入记忆
        
        Args:
            path: 相对路径，如 "conversations/stream_123/2026-03-27.md"
            content: Markdown 内容
            weight: 初始权重（默认 1.0）
            
        Returns:
            完整路径
        """
        full_path = self.base_path / path
        full_path.parent.mkdir(parents=True, exist_ok=True)
        
        async with aiofiles.open(full_path, "w", encoding="utf-8") as f:
            await f.write(content)
        
        # 更新元数据
        meta = await self._load_meta()
        meta[path] = {
            "weight": weight,
            "created": datetime.now().isoformat(),
            "accessed": datetime.now().isoformat(),
            "access_count": 0,
        }
        await self._save_meta(meta)
        
        return str(full_path)
    
    async def read(self, path: str) -> Optional[str]:
        """读取记忆"""
        full_path = self.base_path / path
        
        if not full_path.exists():
            return None
        
        async with aiofiles.open(full_path, "r", encoding="utf-8") as f:
            content = await f.read()
        
        # 强化记忆（被使用 = 被强化）
        await self._strengthen(path)
        
        return content
    
    async def edit(
        self,
        path: str,
        changes: dict,
    ) -> bool:
        """编辑记忆
        
        Args:
            path: 相对路径
            changes: 修改内容
                - append: 追加内容
                - prepend: 前置内容
                - replace: {"old": "new"} 替换
        """
        content = await self.read(path)
        if content is None:
            return False
        
        if "prepend" in changes:
            content = changes["prepend"] + "\n" + content
        
        if "append" in changes:
            content += "\n" + changes["append"]
        
        if "replace" in changes:
            for old, new in changes["replace"].items():
                content = content.replace(old, new)
        
        # 写回（不重置权重）
        full_path = self.base_path / path
        async with aiofiles.open(full_path, "w", encoding="utf-8") as f:
            await f.write(content)
        
        return True
    
    async def delete(self, path: str) -> bool:
        """删除记忆（遗忘）
        
        不是真正删除，而是移动到 archive
        """
        full_path = self.base_path / path
        
        if not full_path.exists():
            return False
        
        # 移动到归档
        archive_path = self.base_path / "archive" / path
        archive_path.parent.mkdir(parents=True, exist_ok=True)
        full_path.rename(archive_path)
        
        # 更新元数据
        meta = await self._load_meta()
        if path in meta:
            del meta[path]
        await self._save_meta(meta)
        
        return True
    
    async def list(
        self,
        directory: str = "",
        pattern: str = "*.md",
    ) -> list[str]:
        """列出记忆"""
        target = self.base_path / directory
        
        if not target.exists():
            return []
        
        paths = []
        for path in target.rglob(pattern):
            # 排除归档和元数据
            if "archive" in path.parts or path.name.startswith("."):
                continue
            rel_path = path.relative_to(self.base_path)
            paths.append(str(rel_path))
        
        return paths
    
    async def search(
        self,
        query: str,
        limit: int = 10,
    ) -> list[dict]:
        """搜索记忆（简单关键词匹配）
        
        完整实现应该使用 BM25 + 向量检索
        """
        results = []
        
        for path in await self.list():
            content = await self.read(path)
            if content and query.lower() in content.lower():
                # 计算简单相关性分数
                score = content.lower().count(query.lower())
                results.append({
                    "path": path,
                    "summary": content[:200] + "..." if len(content) > 200 else content,
                    "score": score,
                })
        
        # 按分数排序
        results.sort(key=lambda x: x["score"], reverse=True)
        
        return results[:limit]
    
    async def history(self, path: str) -> list[dict]:
        """获取记忆历史
        
        简单实现：检查归档目录
        完整实现：应该使用 Git 或版本系统
        """
        archive_path = self.base_path / "archive" / path
        
        if archive_path.exists():
            async with aiofiles.open(archive_path, "r", encoding="utf-8") as f:
                content = await f.read()
            return [{
                "version": "archived",
                "content": content,
            }]
        
        return []
    
    async def _strengthen(self, path: str, delta: float = 0.1) -> None:
        """强化记忆权重"""
        meta = await self._load_meta()
        
        if path in meta:
            meta[path]["weight"] = min(2.0, meta[path]["weight"] + delta)
            meta[path]["accessed"] = datetime.now().isoformat()
            meta[path]["access_count"] = meta[path].get("access_count", 0) + 1
            await self._save_meta(meta)
    
    async def decay(self, decay_rate: float = 0.95) -> int:
        """衰减所有记忆权重
        
        应该定期调用（如每天一次）
        
        Returns:
            衰减的记忆数量
        """
        meta = await self._load_meta()
        count = 0
        
        for path, info in meta.items():
            old_weight = info["weight"]
            new_weight = old_weight * decay_rate
            
            if new_weight < 0.01:
                # 权重过低，移动到归档
                await self.delete(path)
            else:
                info["weight"] = new_weight
                count += 1
        
        await self._save_meta(meta)
        return count
```

### 2.3 事件系统 (event/stream.py)

所有输入统一进入事件流。

```python
"""事件流：统一的输入入口"""

import json
from pathlib import Path
from datetime import datetime
from typing import Optional
from dataclasses import dataclass, asdict
import aiofiles


@dataclass
class Event:
    """事件数据结构"""
    id: str
    type: str  # message, link, reminder, trigger, follow_up, system
    source: str  # 来源（平台/用户/系统）
    content: str
    summary: str
    timestamp: str
    metadata: dict
    
    @classmethod
    def from_dict(cls, data: dict) -> "Event":
        return cls(**data)
    
    def to_dict(self) -> dict:
        return asdict(self)


class EventStream:
    """事件流管理"""
    
    def __init__(self, base_path: str | Path) -> None:
        self.base_path = Path(base_path)
        
        # 创建目录结构
        (self.base_path / "pending").mkdir(parents=True, exist_ok=True)
        (self.base_path / "processed").mkdir(parents=True, exist_ok=True)
        (self.base_path / "archive").mkdir(parents=True, exist_ok=True)
    
    async def push(self, event: Event) -> str:
        """推送事件到待处理队列"""
        path = self.base_path / "pending" / f"{event.id}.json"
        
        async with aiofiles.open(path, "w", encoding="utf-8") as f:
            await f.write(json.dumps(event.to_dict(), indent=2, ensure_ascii=False))
        
        return event.id
    
    async def create_event(
        self,
        type: str,
        source: str,
        content: str,
        summary: Optional[str] = None,
        metadata: Optional[dict] = None,
    ) -> Event:
        """创建并推送事件"""
        event = Event(
            id=f"evt_{datetime.now().strftime('%Y%m%d_%H%M%S_%f')}",
            type=type,
            source=source,
            content=content,
            summary=summary or content[:100],
            timestamp=datetime.now().isoformat(),
            metadata=metadata or {},
        )
        await self.push(event)
        return event
    
    async def has_pending(self) -> bool:
        """检查是否有待处理事件"""
        pending_dir = self.base_path / "pending"
        return any(pending_dir.iterdir())
    
    async def get_pending(self, limit: int = 10) -> list[Event]:
        """获取待处理事件"""
        pending_dir = self.base_path / "pending"
        events = []
        
        # 按时间顺序获取
        files = sorted(pending_dir.glob("*.json"))
        
        for path in files[:limit]:
            async with aiofiles.open(path, "r", encoding="utf-8") as f:
                content = await f.read()
                data = json.loads(content)
                events.append(Event.from_dict(data))
        
        return events
    
    async def mark_processed(self, event_id: str) -> bool:
        """标记事件为已处理"""
        pending_path = self.base_path / "pending" / f"{event_id}.json"
        
        if not pending_path.exists():
            return False
        
        processed_path = self.base_path / "processed" / f"{event_id}.json"
        pending_path.rename(processed_path)
        
        return True
    
    async def get_event(self, event_id: str) -> Optional[Event]:
        """获取特定事件"""
        # 先找 pending
        for dir_name in ["pending", "processed", "archive"]:
            path = self.base_path / dir_name / f"{event_id}.json"
            if path.exists():
                async with aiofiles.open(path, "r", encoding="utf-8") as f:
                    content = await f.read()
                    return Event.from_dict(json.loads(content))
        
        return None
    
    async def list_events(
        self,
        status: str = "pending",
        limit: int = 20,
    ) -> list[Event]:
        """列出事件"""
        dir_path = self.base_path / status
        events = []
        
        if not dir_path.exists():
            return events
        
        files = sorted(dir_path.glob("*.json"), reverse=True)
        
        for path in files[:limit]:
            async with aiofiles.open(path, "r", encoding="utf-8") as f:
                content = await f.read()
                events.append(Event.from_dict(json.loads(content)))
        
        return events
    
    async def archive_old(self, days: int = 1) -> int:
        """归档旧的已处理事件"""
        from datetime import timedelta
        
        processed_dir = self.base_path / "processed"
        count = 0
        cutoff = datetime.now() - timedelta(days=days)
        
        for path in processed_dir.glob("*.json"):
            async with aiofiles.open(path, "r", encoding="utf-8") as f:
                content = await f.read()
                data = json.loads(content)
                event_time = datetime.fromisoformat(data["timestamp"])
                
                if event_time < cutoff:
                    archive_path = self.base_path / "archive" / path.name
                    path.rename(archive_path)
                    count += 1
        
        return count
```
